拖拽合并功能：
  • 拖到左侧元素：
    • 拖到左边缘（左侧33%）：避让交换位置
    • 拖到右边缘和中间：目标元素高亮，松开后合并为 "目标 、拖动"，原元素消失
  • 拖到右侧元素：
    • 拖到右边缘（右侧33%）：避让交换位置
    • 拖到左边缘和中间：目标元素高亮，松开后合并为 "拖动 、目标"，原元素消失
new Sortable(gridDemo, {
    animation: 150,
    ghostClass: 'blue-background-class',
    onMove: function(evt, originalEvent) {
        // 自定义合并和交换逻辑
        // 拖到边缘：避让交换
        // 拖到中间：高亮准备合并
    },
    onEnd: function(evt) {
        // 松开时执行合并操作
    }
});
